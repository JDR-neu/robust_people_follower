132  void calculateVelocity()
133  {
134 
135  // if the current or old position is not specified, the velocity is set to 0
136  if (m_current_pose.pose.position.x == 0.0 || m_old_pose.pose.position.x == 0.0)
137  m_velocity = 0.0;
138 
139  // otherwise, the velocity is calculated by the distance between the last two points divided by the difference
140  // in time
141  else {
142  auto delta_t{(m_current_pose.header.stamp - m_old_pose.header.stamp).toSec()};
143  m_velocity = sqrt(pow((m_old_pose.pose.position.x - m_current_pose.pose.position.x), 2) +
144  pow((m_old_pose.pose.position.y - m_current_pose.pose.position.y), 2)) / delta_t;
145  }
146  }
